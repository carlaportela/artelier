  %% ── Auth.js ──────────────────────────────────────────
  USER {
    string  id                 PK
    string  name
    string  email              UK
    string  password
    enum    role               "ARTISAN | BUYER | ADMIN"
    string  stripeAccountId
    boolean firstSaleCompleted
    string  twoFactorSecret
    boolean twoFactorEnabled
    boolean suspended
    datetime deletedAt
    datetime createdAt
    datetime updatedAt
  }

  ACCOUNT {
    string id        PK
    string userId    FK
    string provider
    string providerAccountId
  }

  SESSION {
    string   id           PK
    string   sessionToken UK
    string   userId       FK
    datetime expires
  }

  VERIFICATION_TOKEN {
    string   identifier
    string   token        UK
    datetime expires
  }

  %% ── Catálogo ──────────────────────────────────────────
  PRODUCT {
    string   id           PK
    string   artisanId    FK
    string   name
    int      priceInCents
    enum     type         "UNIQUE | PERISHABLE | STANDARD"
    enum     status       "ACTIVE | SOLD | EXPIRED"
    datetime expiresAt
    string[] imageUrls
    string   category
    string   locality
    datetime deletedAt
  }

  PROCESS_UPDATE {
    string   id        PK
    string   artisanId FK
    string   content
    string   imageUrl
    datetime deletedAt
    datetime createdAt
  }

  %% ── Pedidos ───────────────────────────────────────────
  ORDER {
    string   id                    PK
    string   buyerId               FK
    string   artisanId             FK
    string   productId             FK
    enum     status                "CONFIRMED | IN_PREPARATION | READY | SHIPPED | DELIVERED | ACCEPTED | CANCELLED | REFUNDED | IN_DISPUTE"
    enum     shippingMethod        "PLATFORM | ARTISAN_OWN | PICKUP"
    int      priceInCents
    int      platformFeeInCents
    int      stripeFeeInCents
    int      totalInCents
    string   stripePaymentIntentId UK
    string   stripeEventId         UK
    string   trackingNumber
    boolean  paidOut
    datetime deletedAt
  }

  %% ── Mensajería ────────────────────────────────────────
  CONVERSATION {
    string   id            PK
    string   participantId FK
    datetime deletedAt
    datetime createdAt
  }

  MESSAGE {
    string   id             PK
    string   conversationId FK
    string   senderId       FK
    string   content
    string   imageUrl
    datetime readAt
    datetime deletedAt
    datetime createdAt
  }

  %% ── Confianza y Disputas ──────────────────────────────
  DISPUTE {
    string   id        PK
    string   orderId   FK
    string   openedBy  FK
    string   reason
    enum     status    "OPEN | RETURN_REQUESTED | RETURN_IN_TRANSIT | ESCALATED | RESOLVED | REJECTED"
    datetime deletedAt
    datetime createdAt
  }

  SEAL {
    string  id          PK
    string  name        UK
    string  type        "PRODUCT | PROFILE"
    boolean isAutomatic
    datetime deletedAt
  }

  PRODUCT_SEAL {
    string   id        PK
    string   productId FK
    string   sealId    FK
    datetime createdAt
  }

  SEAL_REQUEST {
    string   id            PK
    string   artisanId     FK
    string   productId     FK "nullable"
    string   sealId        FK
    enum     status        "PENDING | APPROVED | REJECTED"
    string   justification
    string   rejectReason
    datetime deletedAt
  }

  %% ── Relaciones sociales ───────────────────────────────
  FOLLOW {
    string   id          PK
    string   followerId  FK
    string   followingId FK
    datetime createdAt
  }

  %% ── Auditoría ─────────────────────────────────────────
  TRANSACTION_LOG {
    string   id        PK
    string   orderId
    string   eventType
    json     payload
    datetime createdAt
    %% Sin deletedAt — retención 5 años obligatoria (NFR12)
  }
